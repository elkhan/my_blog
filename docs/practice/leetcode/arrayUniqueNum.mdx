```ts
const arr = [4, 1, 2, 1, 2]

function singleNumber(nums: number[]) {
  return nums.find((x) => nums.indexOf(x) === nums.lastIndexOf(x))
}
```

export const SingleNumber = () => {
  const arr = [4, 1, 2, 1, 2]
  function singleNumber(nums) {
    return nums.find((x) => nums.indexOf(x) === nums.lastIndexOf(x))
  }
  return (
    <button onClick={() => alert(singleNumber(arr))}>Check singleNumber result</button>
  )
}

<SingleNumber />

```ts
function singleNumberXOR(nums: number[]) {

}
```


export const SingleNumberXOR = () => {

  }
  return (
