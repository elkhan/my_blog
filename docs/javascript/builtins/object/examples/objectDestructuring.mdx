```js
import { assert } from 'console'

const user = { name: 'John Doe', age: 34 }
const { name, age } = user

assert(name === 'John Doe')
assert(age === 34)

const HIGH_TEMPERATURES = {
  yesterday: 75,
  today: 77,
  tomorrow: 80
}

const { today } = HIGH_TEMPERATURES
const { tomorrow } = HIGH_TEMPERATURES

assert(today === 77)
assert(tomorrow === 80)

// Assigning values
const { name: userName, age: userAge } = user
assert(userName === 'John Doe')
assert(userAge === 34)
assert(userName === name)
assert(userAge === age)

// Assigning values from nested objects
const userDoe = {
  johnDoe: {
    age: 34,
    email: 'johnDoe@freeCodeCamp.com'
  }
}

const {
  johnDoe: { age: userDoeAge, email: userDoeEmail }
} = userDoe // { johnDoe: { age: 34, email: 'johnDoe@freeCodeCamp.com' } }

assert(userDoeAge === 34)
assert(userDoeEmail === 'johnDoe@freeCodeCamp.com')

const LOCAL_FORECAST = {
  yesterday: { low: 61, high: 75 },
  today: { low: 64, high: 77 },
  tomorrow: { low: 68, high: 80 }
}

const {
  today: { low: lowToday }
} = LOCAL_FORECAST
const {
  today: { high: highToday }
} = LOCAL_FORECAST

assert(lowToday === 64)
assert(highToday === 77)

// Use Destructuring Assignment to Pass an Object as a Function's Parameters
const data = {
  profileName: 'Jake',
  profileAge: 45,
  profileNationality: 'Cosmopolitan',
  profileLocation: 'World'
}
const profileUpdate = (profileData) => {
  const { profileNationality, profileLocation } = profileData
  assert(profileNationality === 'Cosmopolitan')
  assert(profileLocation === 'World')
}
const profileDataDestructured = ({ profileName, profileAge }) => {
  assert(profileName === 'Jake')
  assert(profileAge === 45)
}

profileUpdate(data)
profileDataDestructured(data)

```
