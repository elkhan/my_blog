```js
const userName = 'Joe'
const reception = {
  [`Hey ${userName}`]: 'ok',
}
console.log(reception)
```

---

```js
const items = ['A', 'B', 'C']
const obj = {
  [items]: 'Hello',
}
console.log(obj)
console.log(obj['A,B,C'])
```
