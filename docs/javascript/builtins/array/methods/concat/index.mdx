#### Nested array
```js
const array1 = [1, 2, 3]
const array2 = [1, 2, 3, [4, 5, [6]]]
const array3 = array1.concat(array2)
//outputs [1,2,3,1,2,3,[4,5,[6]]]
```

export const Concat = () => {
  const array1 = [1, 2, 3]
  const array2 = [1, 2, 3, [4, 5, [6]]]
  const array3 = array1.concat(array2)
  return <button onClick={() => alert(JSON.stringify(array3))}>Click me</button>
}

#### Concatenate values to an array
```js
const letters = ["a", "b", "c"];
const alphaNumeric = letters.concat(1, [2, 3])
//outputs ["a","b","c",1,2,3]
```

export const Value = () => {
  const letters = ["a", "b", "c"]
  const numbers = [1, [2, 3]]
  const alphaNumeric = letters.concat(1, [2, 3])
  return <button onClick={() => alert(JSON.stringify(alphaNumeric))}>Click me</button>
}

#### Nested heterogeneous arrays
```js
const letters = ["a", "b", "c"]
const numbers = [1, [2, 3]]

//outputs [ 1, empty, 3, 4, 5 ]
```
